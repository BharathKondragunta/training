function main(m, n)
{
    new Promise((resolve, reject) => {
        setTimeout(()=> {
            if(isNaN(m) || isNaN(n))
            {
                return reject()
            }
            for(i=m; i<=n; i++)
            {
                console.log(i);
            }
            return resolve()
        },5000)
        })
        .then(() => console.log("Promise resolved"))
        .catch(() => console.log("error"))
}
main(1,"10a")
